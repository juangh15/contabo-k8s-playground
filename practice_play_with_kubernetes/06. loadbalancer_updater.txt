# Nginx Updater
cat <<'EOF' > nginx_updater.sh
NGINX_CONF="/etc/nginx/conf.d/nginx.conf"
KUBECONFIG="./kubeconfig-external.yaml"
NODEPORT="30080"
WORKER_IPS=$(. ./get_worker_ips.sh)

if [[ -z "$WORKER_IPS" ]]; then
  echo "No se encontraron nodos worker. Abortando actualizaci√≥n de Nginx."
  exit 1
fi

cat <<NGINX_EOF > $NGINX_CONF
worker_processes 1;
events {
    worker_connections 1024;
}
http {
    upstream backend {
$(echo "$WORKER_IPS" | awk '{print "        server " $1 ":'$NODEPORT';"}')
    }
    server {
        listen 80;
        location / {
            proxy_pass http://backend;
            proxy_set_header Host \$host;
            proxy_set_header X-Real-IP \$remote_addr;
            proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto \$scheme;
        }
    }
}
NGINX_EOF

# Reload
docker exec nginx-lb nginx -s reload
echo "Config updated in nodes:"
echo "$WORKER_IPS"
EOF


# Execution Phase
. nginx_updater.sh
